    <script type="module">
        // --- CONFIG & INITIALIZATION ---
        tailwind.config = {
            theme: {
                extend: {
                    fontFamily: {
                        'display': ['"Playfair Display"', 'serif'],
                        'sans': ['"Satoshi"', 'sans-serif'],
                    },
                }
            }
        };

        document.addEventListener('DOMContentLoaded', () => {
            // --- STATE MANAGEMENT ---
            let currentStep = 1;
            const totalSteps = 5;

            // --- DOM ELEMENTS ---
            const stepBtns = document.querySelectorAll('.step-btn');
            const progressBar = document.getElementById('progress-bar');
            const celebrateBtn = document.getElementById('celebrate-btn');
            const finalMessage = document.getElementById('final-message');
            const polaroid = document.querySelector('.polaroid');
            
            // --- THREE.JS BACKGROUND HEARTS ---
            let scene, camera, renderer, hearts, finalAnimation = false;

            function initThree() {
                scene = new THREE.Scene();
                camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
                renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('three-canvas'), alpha: true });
                renderer.setSize(window.innerWidth, window.innerHeight);

                // Lighting
                const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
                scene.add(ambientLight);
                const directionalLight = new THREE.DirectionalLight(0xffc0cb, 1);
                directionalLight.position.set(5, 5, 5);
                scene.add(directionalLight);

                // Heart Geometry
                const heartShape = new THREE.Shape();
                heartShape.moveTo(0.25, 0.25);
                heartShape.bezierCurveTo(0.25, 0.25, 0.2, 0, 0, 0);
                heartShape.bezierCurveTo(-0.3, 0, -0.3, 0.35, -0.3, 0.35);
                heartShape.bezierCurveTo(-0.3, 0.55, -0.1, 0.77, 0.25, 0.95);
                heartShape.bezierCurveTo(0.6, 0.77, 0.8, 0.55, 0.8, 0.35);
                heartShape.bezierCurveTo(0.8, 0.35, 0.8, 0, 0.5, 0);
                heartShape.bezierCurveTo(0.35, 0, 0.25, 0.25, 0.25, 0.25);
                
                const extrudeSettings = { depth: 0.1, bevelEnabled: true, bevelSegments: 2, steps: 2, bevelSize: 0.05, bevelThickness: 0.05 };
                const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
                const material = new THREE.MeshStandardMaterial({ color: 0xff4f8f, roughness: 0.5, metalness: 0.2 });

                // Instanced Mesh for performance
                const count = 25;
                hearts = new THREE.InstancedMesh(geometry, material, count);
                scene.add(hearts);

                const dummy = new THREE.Object3D();
                for (let i = 0; i < count; i++) {
                    dummy.position.set(
                        (Math.random() - 0.5) * 10,
                        (Math.random() - 0.5) * 10,
                        (Math.random() - 0.5) * 10
                    );
                    dummy.rotation.set(
                        Math.random() * Math.PI,
                        Math.random() * Math.PI,
                        Math.random() * Math.PI
                    );
                    const scale = Math.random() * 0.5 + 0.2;
                    dummy.scale.set(scale, scale, scale);
                    dummy.updateMatrix();
                    hearts.setMatrixAt(i, dummy.matrix);

                    // Store extra data for animation
                    hearts.userData[i] = {
                        initialY: dummy.position.y,
                        randomOffset: Math.random() * Math.PI * 2,
                        rotationSpeed: (Math.random() - 0.5) * 0.01
                    };
                }

                camera.position.z = 5;
                animate();
            }

            function animate() {
                requestAnimationFrame(animate);
                const time = Date.now() * 0.0005;

                if (hearts) {
                    const dummy = new THREE.Object3D();
                    for (let i = 0; i < hearts.count; i++) {
                        hearts.getMatrixAt(i, dummy.matrix);
                        dummy.matrix.decompose(dummy.position, dummy.quaternion, dummy.scale);

                        if (finalAnimation) {
                            dummy.position.y += 0.02;
                            dummy.position.x += Math.sin(time + hearts.userData[i].randomOffset) * 0.01;
                            dummy.rotation.y += hearts.userData[i].rotationSpeed * 5;
                            dummy.scale.multiplyScalar(0.995);
                        } else {
                            dummy.position.y = hearts.userData[i].initialY + Math.sin(time + hearts.userData[i].randomOffset) * 0.5;
                            dummy.rotation.x += hearts.userData[i].rotationSpeed;
                            dummy.rotation.y += hearts.userData[i].rotationSpeed;
                        }

                        dummy.updateMatrix();
                        hearts.setMatrixAt(i, dummy.matrix);
                    }
                    hearts.instanceMatrix.needsUpdate = true;
                }

                renderer.render(scene, camera);
            }

            function onWindowResize() {
                camera.aspect = window.innerWidth / window.innerHeight;
                camera.updateProjectionMatrix();
                renderer.setSize(window.innerWidth, window.innerHeight);
            }

            window.addEventListener('resize', onWindowResize, false);
            initThree();

            // --- ANIMATION & NAVIGATION ---
            function updateProgressBar() {
                const progress = (currentStep - 1) / (totalSteps - 1);
                gsap.to(progressBar, { width: `${progress * 100}%`, duration: 0.5, ease: 'power2.out' });
            }

            function goToStep(step) {
                if (step > totalSteps || step < 1) return;

                const currentCard = document.getElementById(`step-${currentStep}`);
                const nextCard = document.getElementById(`step-${step}`);
                
                const tl = gsap.timeline({
                    onComplete: () => {
                        currentStep = step;
                        updateProgressBar();
                    }
                });

                tl.to(currentCard, { autoAlpha: 0, scale: 0.95, duration: 0.4, ease: 'power2.in' })
                  .set(currentCard, { display: 'none' })
                  .set(nextCard, { display: 'flex', autoAlpha: 0, scale: 1.05, y: 20 })
                  .to(nextCard, { autoAlpha: 1, scale: 1, y: 0, duration: 0.6, ease: 'power2.out' })
                  .from(nextCard.querySelectorAll('.stagger-in'), { autoAlpha: 0, y: 20, stagger: 0.15 }, "-=0.4");
            }

            stepBtns.forEach((btn, index) => {
                if (index < totalSteps - 1) { // All buttons except the last one
                    btn.addEventListener('click', () => {
                        goToStep(currentStep + 1);
                    });
                }
            });

            // --- POLAROID INTERACTION ---
            if (polaroid) {
                polaroid.addEventListener('mousemove', (e) => {
                    const rect = polaroid.getBoundingClientRect();
                    const x = e.clientX - rect.left;
                    const y = e.clientY - rect.top;
                    const rotateX = (y / rect.height - 0.5) * -30; // -15 to 15 degrees
                    const rotateY = (x / rect.width - 0.5) * 30; // -15 to 15 degrees
                    
                    gsap.to(polaroid, {
                        rotationX: rotateX,
                        rotationY: rotateY,
                        rotationZ: -5,
                        duration: 0.5,
                        ease: 'power2.out'
                    });
                });
                polaroid.addEventListener('mouseleave', () => {
                    gsap.to(polaroid, {
                        rotationX: 0,
                        rotationY: 0,
                        rotationZ: -5,
                        duration: 0.5,
                        ease: 'power2.out'
                    });
                });
            }

            // --- GRAND FINALE ---
            celebrateBtn.addEventListener('click', () => {
                gsap.to(celebrateBtn, { autoAlpha: 0, duration: 0.5, onComplete: () => { celebrateBtn.style.display = 'none'; } });

                gsap.to(finalMessage, { autoAlpha: 1, y: -20, duration: 1, ease: 'power3.out', delay: 0.5 });

                startConfetti();
                finalAnimation = true;
            });
            
            function startConfetti() {
                const duration = 5 * 1000;
                const animationEnd = Date.now() + duration;
                const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 100 };

                function randomInRange(min, max) {
                    return Math.random() * (max - min) + min;
                }

                const interval = setInterval(() => {
                    const timeLeft = animationEnd - Date.now();

                    if (timeLeft <= 0) {
                        return clearInterval(interval);
                    }

                    const particleCount = 50 * (timeLeft / duration);
                    // Since it's a special day, let's use some fun shapes
                    confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 } }));
                    confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 } }));
                }, 250);

                // Add emoji confetti
                const emojiConfetti = confetti.create(null, { resize: true });
                (function frame() {
                  if (Date.now() < animationEnd) {
                      emojiConfetti({
                        particleCount: 2,
                        angle: 90,
                        spread: 360,
                        startVelocity: 40,
                        origin: {
                          x: Math.random(),
                          y: Math.random() - 0.2
                        },
                        shapes: ['emoji'],
                        shapeOptions: {
                          emoji: {
                            value: ['‚ù§Ô∏è', 'üíñ', '‚ú®']
                          }
                        },
                        zIndex: 101
                      });
                      requestAnimationFrame(frame);
                  }
                }());
            }

            // --- INITIAL LOAD ANIMATION ---
            const firstCard = document.getElementById('step-1');
            gsap.set(firstCard, { display: 'flex', autoAlpha: 0, scale: 1.05, y: 20 });
            gsap.to(firstCard, { autoAlpha: 1, scale: 1, y: 0, duration: 0.8, ease: 'power2.out', delay: 0.5 });
            gsap.from(firstCard.querySelectorAll('.stagger-in'), { autoAlpha: 0, y: 20, stagger: 0.2, delay: 0.8 });
            updateProgressBar();
        });
    </script>
